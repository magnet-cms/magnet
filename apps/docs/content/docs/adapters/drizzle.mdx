# Drizzle Adapter

The `@magnet-cms/adapter-drizzle` package provides PostgreSQL support for Magnet CMS through [Drizzle ORM](https://orm.drizzle.team). It supports the Neon serverless driver and standard PostgreSQL connections, enabling deployment to serverless environments and managed databases.

## Installation

```bash
bun add @magnet-cms/adapter-drizzle
```

For Neon serverless (recommended for Neon databases):

```bash
bun add @magnet-cms/adapter-drizzle @neondatabase/serverless
```

For standard PostgreSQL:

```bash
bun add @magnet-cms/adapter-drizzle postgres
```

## Important: Adapter Initialization Order

**You must call `setDatabaseAdapter('drizzle')` before importing any schema files.** This is because schema decorators resolve at import time, and the adapter must be set before they execute.

```typescript
// app.module.ts — IMPORTANT: setDatabaseAdapter MUST be first
import { setDatabaseAdapter } from '@magnet-cms/common'
setDatabaseAdapter('drizzle')

// Only import schemas AFTER the adapter is set
import { MagnetModule } from '@magnet-cms/core'
import { Module } from '@nestjs/common'
import { CatsModule } from './modules/cats/cats.module'
```

## Configuration

### Neon (serverless PostgreSQL)

```typescript
import { setDatabaseAdapter } from '@magnet-cms/common'
setDatabaseAdapter('drizzle')

import { MagnetModule } from '@magnet-cms/core'
import { Module } from '@nestjs/common'
import { ConfigModule } from '@nestjs/config'

@Module({
  imports: [
    ConfigModule.forRoot({ isGlobal: true }),
    MagnetModule.forRoot({
      db: {
        connectionString: process.env.DATABASE_URL || '',
        dialect: 'postgresql',
        driver: 'neon',
        debug: process.env.NODE_ENV === 'development',
      },
      jwt: {
        secret: process.env.JWT_SECRET || 'development-secret-key',
      },
    }),
  ],
})
export class AppModule {}
```

### Standard PostgreSQL

```typescript
import { setDatabaseAdapter } from '@magnet-cms/common'
setDatabaseAdapter('drizzle')

import { MagnetModule } from '@magnet-cms/core'
import { Module } from '@nestjs/common'

@Module({
  imports: [
    MagnetModule.forRoot({
      db: {
        connectionString: process.env.DATABASE_URL || 'postgresql://localhost:5432/mydb',
        dialect: 'postgresql',
        debug: false,
      },
      jwt: {
        secret: process.env.JWT_SECRET || 'your-secret-key',
      },
    }),
  ],
})
export class AppModule {}
```

## Configuration Options

| Option | Type | Required | Description |
|--------|------|----------|-------------|
| `connectionString` | `string` | Yes | PostgreSQL connection string (`postgresql://user:pass@host/db`) |
| `dialect` | `'postgresql'` | Yes | Database dialect (currently `postgresql` only) |
| `driver` | `'neon'` \| `undefined` | No | Use `'neon'` for Neon serverless; omit for standard PostgreSQL |
| `debug` | `boolean` | No | Enable SQL query logging |

## Schema Definition

Use the `@Schema`, `@Field.*`, and `@Field.Validators` decorators — the same API as the Mongoose adapter:

```typescript
import { Field, Schema } from '@magnet-cms/common'
import {
  IsBoolean,
  IsDate,
  IsNotEmpty,
  IsNumber,
  IsOptional,
  IsString,
  Length,
  Max,
  Min,
} from 'class-validator'
import { Type } from 'class-transformer'

@Schema()
export class Cat {
  @Field.Text({ unique: true, required: true, tab: 'General', row: true })
  @Field.Validators(IsString(), Length(10, 20), IsNotEmpty())
  tagID: string

  @Field.Text({ required: true, intl: true, tab: 'General', description: 'Cat name' })
  @Field.Validators(IsString(), Length(2, 255), IsNotEmpty())
  name: string

  @Type(() => Date)
  @Field.Date({ required: true, tab: 'General' })
  @Field.Validators(IsDate(), IsNotEmpty())
  birthdate: Date

  @Field.Number({ required: true, tab: 'General', description: 'Weight in kg' })
  @Field.Validators(IsNumber(), Min(0.5), Max(15), IsNotEmpty())
  weight: number

  // Relation to another schema (stored as UUID string)
  @Field.Relationship({ ref: 'Owner', required: true, tab: 'Relations' })
  @Field.Validators(IsNotEmpty())
  owner!: string

  @Field.Boolean({ required: true, tab: 'General' })
  @Field.Validators(IsBoolean())
  castrated: boolean

  @Field.Text({ required: false, intl: true, tab: 'General' })
  @Field.Validators(IsString(), IsOptional())
  description?: string
}
```

## Service Usage

Inject the model using `@InjectModel` from `@magnet-cms/common`:

```typescript
import { Injectable } from '@nestjs/common'
import { InjectModel, Model } from '@magnet-cms/common'
import { Cat } from './cat.schema'

@Injectable()
export class CatsService {
  constructor(@InjectModel(Cat) private model: Model<Cat>) {}

  async findAll() {
    return this.model.find()
  }

  async findById(id: string) {
    return this.model.findById(id)
  }

  async create(data: Partial<Cat>) {
    return this.model.create(data)
  }

  async update(id: string, data: Partial<Cat>) {
    return this.model.update({ id }, data)
  }

  async delete(id: string) {
    return this.model.delete({ id })
  }
}
```

## Query Builder

```typescript
// Filter with conditions
const cats = await this.model
  .query()
  .where({ castrated: true })
  .exec()

// Sort and paginate
const page = await this.model
  .query()
  .sort({ name: 1 })
  .skip(0)
  .limit(10)
  .exec()

// Get paginated result with totals
const result = await this.model
  .query()
  .where({ castrated: false })
  .paginate()
// Returns: { data: [...], total: 42, page: 1, limit: 10 }
```

## Internationalization (i18n)

Enable per-field i18n with `intl: true`:

```typescript
@Schema({ i18n: true })
export class Article {
  @Field.Text({ required: true, intl: true })
  title: string  // Stored per locale

  @Field.Text({ required: true })
  slug: string   // Shared across locales
}
```

## Module Setup

Register your schema module with NestJS:

```typescript
import { Module } from '@nestjs/common'
import { MagnetModule } from '@magnet-cms/core'
import { Cat } from './cat.schema'
import { CatsController } from './cats.controller'
import { CatsService } from './cats.service'

@Module({
  imports: [MagnetModule.forFeature([Cat])],
  controllers: [CatsController],
  providers: [CatsService],
  exports: [CatsService],
})
export class CatsModule {}
```

## Differences from Mongoose Adapter

| Feature | Drizzle Adapter | Mongoose Adapter |
|---------|----------------|-----------------|
| Database | PostgreSQL | MongoDB |
| IDs | UUID strings | ObjectId strings |
| Relations | UUID foreign keys | ObjectId references |
| Driver | `neon` or standard `postgres` | Mongoose ODM |
| `setDatabaseAdapter` | **Required** (call first) | Not needed |
| Serverless | Supported (Neon driver) | Not recommended |
