# Quick Start

This guide walks you through creating your first content schema with Magnet.

## Creating a Schema

<Steps>
  <Step>
    ### Define your schema

    Create a new file `src/schemas/post.schema.ts`:

    ```ts
    import { Schema, Prop, UI, Validators } from '@magnet/common'
    import { Document } from 'mongoose'

    @Schema({
      collection: 'posts',
      displayName: 'Blog Posts',
      description: 'Blog post content type',
    })
    export class Post extends Document {
      @Prop({ required: true })
      @UI({ label: 'Title', placeholder: 'Enter post title' })
      @Validators({ minLength: 3, maxLength: 200 })
      title: string

      @Prop({ required: true })
      @UI({ label: 'Slug', description: 'URL-friendly identifier' })
      slug: string

      @Prop()
      @UI({ label: 'Content', component: 'richtext' })
      content: string

      @Prop({ default: false })
      @UI({ label: 'Published', component: 'switch' })
      published: boolean

      @Prop({ default: Date.now })
      @UI({ label: 'Created At', readOnly: true })
      createdAt: Date
    }
    ```
  </Step>
  <Step>
    ### Register the schema

    Update your module to include the schema:

    ```ts
    import { Module } from '@nestjs/common'
    import { MagnetModule } from '@magnet/core'
    import { MongooseAdapter } from '@magnet/adapter-mongoose'
    import { Post } from './schemas/post.schema'

    @Module({
      imports: [
        MagnetModule.forRoot({
          adapter: new MongooseAdapter({
            uri: process.env.MONGODB_URI,
          }),
          schemas: [Post],
        }),
      ],
    })
    export class AppModule {}
    ```
  </Step>
  <Step>
    ### Access the Admin UI

    Start your server and navigate to:

    ```
    http://localhost:3000/admin
    ```

    You'll see your Post schema in the content manager.
  </Step>
</Steps>

## Understanding Decorators

### @Schema

Defines schema-level metadata:

```ts
@Schema({
  collection: 'posts',      // MongoDB collection name
  displayName: 'Blog Posts', // Human-readable name
  description: 'Description shown in admin',
  icon: 'file-text',        // Icon for admin UI
})
```

### @Prop

Defines property configuration:

```ts
@Prop({
  required: true,    // Field is required
  default: 'value',  // Default value
  unique: true,      // Unique constraint
  index: true,       // Create index
})
```

### @UI

Configures admin UI appearance:

```ts
@UI({
  label: 'Display Label',
  placeholder: 'Placeholder text',
  component: 'richtext', // UI component type
  hidden: false,         // Hide from admin
  readOnly: true,        // Read-only field
})
```

### @Validators

Adds validation rules:

```ts
@Validators({
  minLength: 3,
  maxLength: 200,
  pattern: '^[a-z-]+$',
  custom: (value) => value.length > 0,
})
```
