# Content Module

The Content module provides CRUD operations for managing content.

## Content Service

```ts
import { Injectable } from '@nestjs/common'
import { ContentService } from '@magnet/core'

@Injectable()
export class PostsService {
  constructor(private contentService: ContentService) {}

  async findAll() {
    return this.contentService.findAll('posts')
  }

  async findOne(id: string) {
    return this.contentService.findOne('posts', id)
  }

  async create(data: any) {
    return this.contentService.create('posts', data)
  }

  async update(id: string, data: any) {
    return this.contentService.update('posts', id, data)
  }

  async delete(id: string) {
    return this.contentService.delete('posts', id)
  }
}
```

## Query Options

### Filtering

```ts
// Find published posts
const posts = await contentService.findAll('posts', {
  filter: { published: true },
})
```

### Sorting

```ts
// Sort by creation date
const posts = await contentService.findAll('posts', {
  sort: { createdAt: -1 },
})
```

### Pagination

```ts
// Paginate results
const posts = await contentService.findAll('posts', {
  skip: 0,
  limit: 10,
})
```

### Selecting Fields

```ts
// Select specific fields
const posts = await contentService.findAll('posts', {
  select: ['title', 'slug', 'createdAt'],
})
```

## Hooks

Content operations support lifecycle hooks:

```ts
// Before create hook
contentService.beforeCreate('posts', async (data) => {
  data.slug = generateSlug(data.title)
  return data
})

// After create hook
contentService.afterCreate('posts', async (doc) => {
  await sendNotification(doc)
})
```
